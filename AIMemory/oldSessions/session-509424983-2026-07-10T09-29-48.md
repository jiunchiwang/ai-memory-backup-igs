```typescript
// commands/status.ts
async function handleStatus(ctx: Context) {
  const active = listActiveTasks();
  const statusUrl = `http://localhost:${STATUS_PORT}`;

    }
  });
}
```

**`web_app` 按鈕注意事項**：
- Telegram Mini App 要求 HTTPS URL（`web_app.url` 必須 `https://`）
- localhost 在桌面版 Telegram 可以用 HTTP，但手機版不行
- 解法：偵測環境，如果沒有 HTTPS → fallback 回純文字摘要 + 提示使用者在瀏覽器開 `localhost:3847`

```typescript
const useWebApp = !!process.env.STATUS_HTTPS_URL;
const webAppUrl = process.env.STATUS_HTTPS_URL || statusUrl;

if (useWebApp) {
  // Mini App 按鈕
  await ctx.reply(text, {
    reply_markup: {
      inline_keyboard: [[
        { text: "🖥 開啟監控", web_app: { url: webAppUrl } }
      ]]
    }
  });
} else {
  // Fallback：純文字 + URL
  await ctx.reply(`${text}\n\n🔗 ${statusUrl}`);
}
```

新增 env var：`STATUS_HTTPS_URL`（可選），設了就用 Mini App 按鈕，沒設就 fallback。
